            { id: 2, icon: '🚚', title: 'Cargar Camión', shortDesc: 'Logística física', fullDesc: 'Inicio de logística física y despacho. El personal de planta procede a cargar la unidad asignada para el traslado.', status: 'normal' },

            { id: 4, icon: '🛣️', title: 'Envío a APT', shortDesc: 'Tránsito al almacén', fullDesc: 'Unidad física en tránsito hacia el almacén de producto terminado. Durante esta fase, el sistema debe reflejar el estatus de "En Tránsito".', status: 'normal' },
            { id: 5, icon: '📱', title: 'Pistoleo', shortDesc: 'Registro en sistema', fullDesc: 'Recepción y registro en sistema mediante dispositivos móviles (pistoleo). Esto asegura la transferencia digital exacta de la custodia de la mercancía.', status: 'success' },

        const mejorasData = [
            { id: 'm1', icon: '📝', title: '1. Checklist Digital en Planta', desc: 'Sustituir la revisión visual manual por un formulario estructurado en tablet. El sistema bloqueará automáticamente el proceso de despacho si falta confirmación de documentación adjunta.' },
            { id: 'm2', icon: '🔔', title: '2. Alertas Automáticas (Tránsito)', desc: 'Implementar notificación en tiempo real, enviada directamente al jefe de almacén APT y supervisores, en el momento exacto en que una unidad sale de planta, generando visibilidad total.' },
            { id: 'm3', icon: '🏷️', title: '3. Estandarización de Etiquetas', desc: 'Asegurar la calidad de impresión y el formato para que el ticket de producción final sea 100% legible para todos los escáneres del sistema en APT, evitando digitación manual.' }
        ];

        function initNav() {
            const buttons = document.querySelectorAll('.tab-btn');
            buttons.forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const targetId = e.target.getAttribute('data-target');
                    
                    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
                    e.target.classList.add('active');
                    
                    document.querySelectorAll('main > section').forEach(sec => sec.classList.add('hidden-section'));
                    document.getElementById(targetId).classList.remove('hidden-section');
                    
                    appState.currentView = targetId;
                    
                    if (targetId === 'view-dashboard' && !appState.chartInstance) {
                        renderChart();
                    }
                });
            });
        }

        function renderFlowTimeline() {
            const container = document.getElementById('flow-container');
            container.innerHTML = '';
            
            flowData.forEach(step => {
                const isCritical = step.status === 'critical';
                const baseClasses = 'flow-step bg-white p-4 rounded-xl border border-slate-200 flex flex-col items-center text-center';
                const criticalClasses = isCritical ? 'critical border-red-300' : '';
                
                const cardHtml = document.createElement('div');
                cardHtml.className = `${baseClasses} ${criticalClasses}`;
                cardHtml.onclick = () => selectFlowStep(step.id, cardHtml);
                
                let numColor = 'bg-blue-500';
                if(step.status === 'critical') numColor = 'bg-red-500 pulse-alert';
                if(step.status === 'success') numColor = 'bg-emerald-500';

                cardHtml.innerHTML = `
                    <div class="w-10 h-10 ${numColor} text-white rounded-full flex items-center justify-center font-bold mb-3 text-lg shadow-sm">${step.id}</div>
                    <div class="text-2xl mb-2">${step.icon}</div>
                    <strong class="text-sm text-slate-800 mb-1 leading-tight">${step.title}</strong>
                    <span class="text-xs text-slate-500">${step.shortDesc}</span>
                `;
                container.appendChild(cardHtml);
            });
        }

        function selectFlowStep(id, element) {
            document.querySelectorAll('.flow-step').forEach(el => el.classList.remove('active'));
            element.classList.add('active');
            
            const stepData = flowData.find(s => s.id === id);
            const panel = document.getElementById('flow-detail-panel');
            
            let badgeHtml = '';
            let borderClass = 'border-l-4 border-blue-500';
            
            if (stepData.status === 'critical') {
                badgeHtml = '<span class="bg-red-100 text-red-800 text-xs px-2 py-1 rounded font-bold uppercase tracking-wider">Control Crítico</span>';
                borderClass = 'border-l-4 border-red-500 bg-red-50';
            } else if (stepData.status === 'success') {
                badgeHtml = '<span class="bg-emerald-100 text-emerald-800 text-xs px-2 py-1 rounded font-bold uppercase tracking-wider">Sistema Actualizado</span>';
                borderClass = 'border-l-4 border-emerald-500';
            }

            panel.innerHTML = `
                <div class="animate-fade-in ${borderClass} pl-6 py-2 -ml-6">
                    <div class="flex items-center gap-4 mb-4">
                        <div class="text-4xl bg-white p-3 rounded-xl shadow-sm border border-slate-100">${stepData.icon}</div>
                        <div>
                            <div class="flex items-center gap-3 mb-1">
                                <h3 class="text-2xl font-bold text-slate-800">Paso ${stepData.id}: ${stepData.title}</h3>
                                ${badgeHtml}
                            </div>
                            <p class="text-slate-500 font-medium">${stepData.shortDesc}</p>
                        </div>
                    </div>
                    <div class="bg-white rounded-lg p-5 border border-slate-200 mt-4 text-slate-700 leading-relaxed text-lg">
                        ${stepData.fullDesc}
                    </div>
                </div>
            `;
        }

        function renderMejoras() {
            const container = document.getElementById('mejoras-container');
            mejorasData.forEach(item => {
                const html = document.createElement('div');
                html.className = 'bg-white border border-slate-200 rounded-xl overflow-hidden shadow-sm hover:shadow-md transition-shadow cursor-pointer';
                html.innerHTML = `
                    <div class="p-5 flex items-start gap-4" onclick="this.nextElementSibling.classList.toggle('hidden-section')">
                        <span class="text-3xl bg-slate-50 p-2 rounded-lg border border-slate-100">${item.icon}</span>
                        <div class="flex-grow pt-1">
                            <h3 class="text-lg font-bold text-slate-800">${item.title}</h3>
                            <p class="text-sm text-slate-500 mt-1">Clic para expandir detalles operativos.</p>
                        </div>
                        <span class="text-slate-400 mt-2">➕</span>
                    </div>
                    <div class="hidden-section bg-slate-50 p-5 border-t border-slate-100 text-slate-700">
                        ${item.desc}
                    </div>
                `;
                container.appendChild(html);
            });
        }

        function renderChart() {
            const ctx = document.getElementById('impactChart').getContext('2d');
            appState.chartInstance = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: ['Despachos Exitosos (Con Doc)', 'Retenidos/Rechazados (Falta Doc)'],
                    datasets: [{
                        data: [82, 18],
                        backgroundColor: [
                            '#10b981', 
                            '#ef4444'  
                        ],
                        borderWidth: 0,
                        hoverOffset: 4
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    cutout: '65%',
                    plugins: {
                        legend: {
                            position: 'bottom',
                            labels: { padding: 20, font: { family: 'Inter', size: 12 } }
                        },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    return ' ' + context.label + ': ' + context.parsed + '%';
                                }
                            },
                            bodyFont: { family: 'Inter' },
                            titleFont: { family: 'Inter' }
                        }
                    }
                }
            });
        }

        document.addEventListener('DOMContentLoaded', () => {
            initNav();
            renderChart();
            renderFlowTimeline();
            renderMejoras();
            
            const style = document.createElement('style');

            document.head.appendChild(style);
        });
